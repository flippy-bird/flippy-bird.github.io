### 代码部分

#### Vertex shader

```OpenGL
#version 430 core
layout(location = 0) in vec3 pos;
layout(location = 1) in vec3 norm;
layout(location = 2) in vec2 tex;
layout(location = 5) in ivec4 boneIds;     //  一个顶点最多收到 4个bone影响
layout(location = 6) in vec4 weights;      // 因此，对应4个权重

uniform mat4 projection;
uniform mat4 view;
uniform mat4 model;

const int MAX_BONES = 100;
const int MAX_BONE_INFLUENCE = 4;
uniform mat4 finalBoneMatrices[MAX_BONES];   // 对应于上面数据结构中的转换矩阵

out vec2 TexCoords;

void main()
{
    vec4 totalPosition = vec4(0.0f);
    for (int i = 0; i < MAX_BONE_INFLUENCE; i++)
    {
        if (boneIds[i] == -1)
            continue;
        if (boneIds[i] >= MAX_BONES)
        {
            totalPosition = vec4(pos, 1.0f);
            break;
         }
         vec4 localPosition = finalBoneMatrices[bondIds[i]] * vec4(pos, 1.0f);
         totalPosition += localPosition * weights[i];
         vec3 localNormal = mat3(finalBonesMatrices[boneIds[i]]) * norm;
     }
     
     mat4 viewModel = view * model;
     gl_position = projection * viewModel * totalPosition;
     TexCoords = tex;
     
 }
```
